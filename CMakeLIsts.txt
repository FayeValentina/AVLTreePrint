cmake_minimum_required(VERSION 3.10)
project(find_package_learning)
set(OpenCV_DIR "E:/opencv/build")
set(BOOST_ROOT "E:/boostLibrary")
set(BOOST_INCLUDEDIR "E:/boostLibrary/include/boost-1_82")
set(BOOST_LIBRARYDIR "E:/boostLibrary/lib")

set(Boost_USE_STATIC_LIBS OFF)        # 只使用动态库
set(Boost_USE_MULTITHREADED ON)       # 启用多线程
set(Boost_USE_STATIC_RUNTIME OFF)     # 使用动态 C++ 运行时

find_package(Boost 1.82 REQUIRED COMPONENTS date_time filesystem iostreams)
message(STATUS "Boost_INCLUDE_DIRS = ${Boost_INCLUDE_DIRS}")
message(STATUS "Boost_LIBRARY_DIRS = ${Boost_LIBRARY_DIRS}")
message(STATUS "Boost_LIBRARIES = ${Boost_LIBRARIES}")
message(STATUS "Boost_VERSION = ${Boost_VERSION}")
find_package(OpenCV REQUIRED)
add_executable(opencv_test main.cpp)  
target_link_libraries(opencv_test ${OpenCV_LIBS} ${Boost_LIBRARIES})
target_include_directories(opencv_test PRIVATE ${OpenCV_INCLUDE_DIRS} ${PROJECT_SOURCE_DIR}/include ${Boost_INCLUDE_DIRS})
